country_event = {

	immediate = {log = "[Root.GetName]: event rivercity.2"}#Auto-logging
	title = rivercity.2.t
	desc = rivercity.2.d
	picture = GFX_event_river_city

	is_triggered_only = yes

	option = {
		name = rivercity.2.a
		add_stability = -0.2
		add_war_support = 0.2
		add_ideas = partial_economic_mobilisation
		add_ideas = limited_conscription
		set_politics = { ruling_party = communism }
		drop_cosmetic_tag = yes
		set_cosmetic_tag = RCT_communism
		hidden_effect = { country_event = rivercity.3 }
		add_popularity = {
			ideology = communism
			popularity = 0.3
		}
	}
}

	id = rivercity.52
	immediate = {log = "[Root.GetName]: event rivercity.52"}#Auto-logging
	title = rivercity.52.t
	desc = rivercity.52.d
	picture = GFX_report_event_physics_lab_01
	
	is_triggered_only = yes
	
	option = {
		name = rivercity.52.a	#sounds good
		FROM = {
			country_event = { days = 1 id = rivercity.53 }
			add_opinion_modifier = {
				target = ROOT
				modifier = RCT_LCT_Pact
			}
		}
		# REVISIT Balance research bonus
		add_tech_bonus = {
			ahead_reduction = 1
			category = cat_light_armor
		}
		add_tech_bonus = {
			bonus = 0.5
			uses = 1
			category = infantry_weapons
		}
	}
	
	option = {
		name = rivercity.52.b #no go away
		ai_chance = { factor = 0 }
		add_political_power = -10
		FROM = {
			country_event = { days = 1 id = rivercity.54 }
		}
	}
}
