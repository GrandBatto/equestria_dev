		completion_reward = {
			
			if = {
				limit = {
					has_idea = poor_science_base
				}
				add_ideas = better_science_base
			}
			else_if = {
				limit = {
					has_idea = better_science_base
				}
				add_ideas = big_science_base
			}
			else_if = {
				limit = {
					has_idea = big_science_base
				}
				add_ideas = proper_science_base
			}
			add_research_slot = 1
			decrease_illiteracy = yes
		}
	}
